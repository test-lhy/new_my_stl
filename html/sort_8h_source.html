89 void TimSort(NormIterator<T> start, NormIterator<T> end, const CmpType<T>& compare_function) {
90  if (end - start <= 64) {
91  Sort(start, end, SortType::INSERT_SORT, compare_function);
92  return;
93  }
94  size_t minrun = GetMinRun(end - start);
95  Index now_index = 0;
96  Index start_index = 0;
97  bool up = true;
98  list<std::pair<Index, Index>> runs;
99  while (now_index <= end - start) {
100  if (now_index - start_index == 1) {
101  if (compare_function(start[now_index], start[start_index])) {
102  up = false;
103  }
104  }
105  if (now_index == end - start ||
106  ((now_index > start_index + 1) && (((up && compare_function(start[now_index], start[now_index])) ||
107  (!up && !compare_function(start[now_index], start[now_index])))))) {
108  while (now_index < end - start && (now_index - start_index) <= minrun) {
109  now_index++;
110  }
111  if (up) {
112  Sort(start + start_index, start + now_index, SortType::INSERT_SORT, compare_function);
113  } else {
114  Sort(start + start_index, start + now_index, SortType::INSERT_SORT,
115  [compare_function](const T& a, const T& b) -> bool { return !compare_function(a, b); });
116  reverse(start + start_index, start + now_index);
117  }
118  runs.push_back({start_index, now_index});
119  if (runs.size() >= 2) {
120  TimSortUpdate(start, runs, compare_function, false);
121  }
122  start_index = now_index;
123  }
124  now_index++;
125  }
126  if (runs.size() >= 2) {
127  TimSortUpdate(start, runs, compare_function, true);
128  }
129 }

147 void TournamentSort(NormIterator<T> start, NormIterator<T> end, const CmpType<T>& compare_function, const T& INF) {
148  size_t front_size = pow(2, ceil(log2(end - start))) - 1;
149  auto* temp_array = new std::pair<T, int>[front_size + end - start];
150  for (Index i = 0; i < end - start; ++i) {
151  temp_array[front_size + i] = {start[i], i};
152  }
153  for (Index i = front_size - 1; i >= 0; --i) {
154  GetWinner(temp_array, i, front_size + end - start, compare_function, INF);
155  }
156  start[0] = temp_array[0].first;
157  Index refresh_index = temp_array[0].second + front_size;
158  temp_array[refresh_index].second = -1;
159  for (int i = 1; i < end - start; ++i) {
160  refresh_index = (refresh_index - 1) / 2;
161  while (refresh_index != 0) {
162  GetWinner(temp_array, refresh_index, front_size + end - start, compare_function, INF);
163  refresh_index = (refresh_index - 1) / 2;
164  }
165  GetWinner(temp_array, refresh_index, front_size + end - start, compare_function, INF);
166  start[i] = temp_array[0].first;
167  refresh_index = temp_array[0].second + front_size;
168  temp_array[refresh_index].first = INF;

316  NormIterator<T> temp_array = new T[end - start];

318  maxT = std::max(*element, maxT);

323  cnt[i - minT] = 0;

329  cnt[i] += cnt[i - 1];

386 T GetApproximateMedian(
387  NormIterator<T> start, NormIterator<T> end,
388  const CmpType<T>& compare_function = [](const T& a, const T& b) -> bool { return a < b; },
389  size_t median_split_length = 5) {
390  for (Index i = 0; i < end - start; i += median_split_length) {
391  Sort(start + i, start + std::min(i + median_split_length, end - start), SortType::INSERT_SORT, compare_function);
392  std::swap(start[i / median_split_length], start[std::min(i + median_split_length / 2, end - start - 1)]);
393  }
394  if ((end - start) <= median_split_length) {
395  return start[0];
396  } else {
397  return GetApproximateMedian(start, start + (end - start) / median_split_length + 1, compare_function,
398  median_split_length);
